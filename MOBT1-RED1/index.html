                public class StartingFile extends Form implements HandlesEventDispatching {

                    private
                    
                    VerticalArrangement mainArrangement;
                    WebViewer bunny;
                    Button rf;
                    protected void $define() {
                
                        this.Sizing("Responsive");
                        mainArrangement=new VerticalArrangement(this);
                        mainArrangement.WidthPercent(100);
                        mainArrangement.Height(LENGTH_FILL_PARENT);
                
                        rf = new Button(mainArrangement);
                        rf.Width(LENGTH_FILL_PARENT);
                        rf.HeightPercent(5);
                        rf.Text("Refresh");
                
                        bunny = new WebViewer(mainArrangement);
                        bunny.HomeUrl("https://t.fachtnaroe.net/red/1/bmi.html");
                        bunny.GoHome();
                
                        
                        EventDispatcher.registerEventForDelegation(this, formName, "Click");
                        
                    }
                
                    public boolean dispatchEvent(Component component, String componentName, String eventName, Object[] params) {
                
                        System.err.print("dispatchEvent: " + formName + " [" + component.toString() + "] [" + componentName + "] " + eventName);
                        if (eventName.equals("Click")) {
                            bunny.GoHome();
                            return true;
                            /*
                            if (component.equals(btnLogin)) {
                                // do something
                            }
                            else if (component.equals(btnRegister)) {
                                switchForm("RegistrationScreen");
                            }
                            */
                        }
                        // true means event has been handled by us (ie recognised)
                        return false;
                    }
                
                }

                function calc() {
                    let answerbox = document.getElementById("answerbox");
                    let weight = document.getElementById("weight").value;
                    let height = document.getElementById("height").value;
                    let answer = weight / ( height * height);
                    answer = -1;
                    if ( (weight != 0) || (height !=0) ) {
                      result = weight / (height * height);
                    }
                    document.getElementById("resultbmi").value = result;
              
          
                    if (result == -1) {
                      answerbox.innerHTML = "Error";
                      
                    }
                    else if (result < 18.5) {
                       answerbox.innerHTML = "Underweight";
                       document.getElementById("answerbox")
                       .style.backgroundColor="blue";
                    }
                    else if (result < 24.9) {
                       answerbox.innerHTML = "Normal";
                       document.getElementById("answerbox")
                       .style.backgroundColor="green";
                    }
                    else if (result < 29.9) {
                       answerbox.innerHTML = "Overweight";
                       document.getElementById("answerbox")
                       .style.backgroundColor="orange";
                    }
                    else  {
                       answerbox.innerHTML = "Obese";
                       document.getElementById("answerbox")
                       .style.backgroundColor="red";
                    }
                  }
